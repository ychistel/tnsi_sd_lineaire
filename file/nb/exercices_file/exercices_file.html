def creer_file():
    return []

def est_vide(file):
    return file == []

def enfiler(file,elt):
    file.append(elt)
    return file

def defiler(file):
    if not est_vide(file):
        tete = file.pop(0)
    return tete

# On ajoute 2 fonctions

def tete_de_file(file):
    if not est_vide(file):
        return file[0]

def longueur(file):
    return len(file)

Suite de Conway

  • En savoir plus sur la suite de Conway

  • Calculer les termes de la suite

[66]:

def terme_en_file(terme):
    f1 = creer_file()
    for chiffre in terme:
        f1 = enfiler(f1, eval(chiffre))
    return f1

def enumere_terme(f1):
    f2 = creer_file()
    if not est_vide(f1):
        c = defiler(f1)
        n = 1
        while not est_vide(f1):
            tete = defiler(f1)
            if tete == c:
                n += 1
            else:
                f2 = enfiler(f2,(n,c))
                c = tete
                n = 1
        f2 = enfiler(f2,(n,c))
    return f2

def terme_suivant(f):
    terme = ''
    while not est_vide(f):
        n,c = defiler(f)
        terme += str(n) + str(c)
    return terme

def suite_conway(n):
    termes = ["1"]
    u = termes[0]
    for i in range(n):
        f1 = terme_en_file(u)
        f2 = enumere_terme(f1)
        u = terme_suivant(f2)
        termes.append(u)
    return termes
