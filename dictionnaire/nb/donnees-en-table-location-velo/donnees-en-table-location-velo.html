from math import sin, cos, acos, pi

def distance(p1, p2):
    """Retourne la distance en mètres entre les 2 points A et B connus grâce à
       leurs coordonnées GPS (en radians).
    """
    latA=p1[0]/180*pi
    longA=p1[1]/180*pi
    latB=p2[0]/180*pi
    longB=p2[1]/180*pi
    # Rayon de la terre en mètres (sphère IAG-GRS80)
    RT = 6378137
    # angle en radians entre les 2 points
    S = acos(sin(latA)*sin(latB) + cos(latA)*cos(latB)*cos(abs(longB-longA)))
    # distance entre les 2 points, comptée sur un arc de grand cercle
    return S*RT

#
# on calcule $x = lat\_2 - lat\_1$
# ensuite on calcule $y = (long\_2 - long\_1) \times \cos((lat\_2 - lat\_1) \times \np{0,0087})$
# et pour finir $\sqrt{x^{2}+y^{2}}$


def distance(p1,p2):
    print(p1,p2)
    x = (p2[1]-p1[1])*cos((p2[0]+p1[0])/2)
    y = p2[0]-p1[0]
    z = sqrt(x**2+y**2)
    print(x,y,z)
    return z*1852*60
