def affiche_labyrinthe(lab):
    n = len(lab)
    m = len(lab[0])
    grid = BlockGrid(m,n)
    grid.lines_on = True
    for i in range(n):
        for j in range(m):
            if lab[i][j] != 1:
                grid[i,j] = [255,255,255]
            else:
                grid[i,j] = [0,0,0]
    for case in solution(lab):
        grid[case] = [255,0,0]
    grid.show()

def est_valide(i,j,lab):
    n = len(lab)
    m = len(lab[0])
    return 0 <= i <n and 0 <= j < m

def voisines(i,j,lab):
    n = len(lab)
    m = len(lab[0])
    v = []
    if lab[i][j] != 1:
        if est_valide(i-1,j,lab) and lab[i-1][j] != 1:
            v.append((i-1,j))
        if est_valide(i+1,j,lab) and lab[i+1][j] != 1:
            v.append((i+1,j))
        if est_valide(i,j-1,lab) and lab[i][j-1] != 1:
            v.append((i,j-1))
        if est_valide(i,j+1,lab) and lab[i][j+1] != 1:
            v.append((i,j+1))
    return v

def depart(lab):
    n = len(lab)
    m = len(lab[0])
    for i in range(n):
        for j in range(m):
            if lab[i][j] == 2:
                return i,j

def arrivee(lab):
    n = len(lab)
    m = len(lab[0])
    for i in range(n):
        for j in range(m):
            if lab[i][j] == 3:
                return i,j

def nb_cases_vides(lab):
    n = len(lab)
    m = len(lab[0])
    nb = 0
    for i in range(n):
        for j in range(m):
            if lab[i][j] != 1:
                nb += 1
    return nb

def solution(lab):
    chemin = [depart(lab)]
    visitees = [depart(lab)]
    case = chemin[0]
    i = case[0]
    j = case[1]
    while lab[i][j] != 3 :
        i = chemin[-1][0]
        j = chemin[-1][1]
        voisins = voisines(i,j,lab)
        for v in visitees:
            if v in voisins:
                voisins.remove(v)
        if len(voisins) > 0:
            case = voisins.pop(0)
            if case not in visitees:
                chemin.append(case)
                visitees.append(case)
        else:
            del chemin[-1]
    chemin.append(arrivee(lab))
    return chemin
